# 招聘数据采集说明文档

> 本文档面向数据采集人员，说明采集招聘数据时的字段格式、采集方法和注意事项。

## 二、字段说明与优先级

### 🔴 必须采集（缺一不可）

| 字段 | 类型 | 说明 | 采集建议 |
|------|------|------|---------|
| `job_title` | string | 岗位标题 | 完整保留，不要截断，如"高级AI算法工程师" |
| `job_description` | string | 岗位描述 | **越完整越好**，必须包含"任职要求"部分 |

> 这两个字段决定图谱能否建立节点，且LLM会从description中推断技能。

### 🟠 强烈建议采集（影响匹配质量）

| 字段 | 类型 | 说明 | 采集建议 |
|------|------|------|---------|
| `skills` | array[string] | 技能列表 | 从"任职要求"中提取，如 `["Python","PyTorch"]` |
| `salary_min` | int | 最低薪资(K/月) | 单位：K/月，如25表示2.5万/月 |
| `salary_max` | int | 最高薪资(K/月) | 单位：K/月 |
| `location` | string | 工作地点 | 城市级别，如"北京"、"上海" |
| `experience_min` | int | 最低经验要求(年) | 年数，如3 |
| `experience_max` | int | 最高经验要求(年) | 年数，如5 |

### 🟡 有就更好的字段

| 字段 | 类型 | 说明 |
|------|------|------|
| `company_name` | string | 公司名称 |
| `education` | string | 学历要求（本科/硕士/博士） |
| `industry` | string | 所属行业 |
| `company_size` | string | 公司规模 |
| `source_url` | string | 原始链接（用于溯源） |
| `source` | string | 数据来源标记，如 `boss_zhipin` / `lagou` / `liepin` |
| `collected_at` | string | 采集时间，ISO格式 |

```python
def extract_skills(description: str) -> list[str]:
    """从岗位描述中提取技能关键词"""
    if not description:
        return []
    desc_lower = description.lower()
    skills = []
    for skill in SKILL_KEYWORDS:
        # 大小写不敏感匹配
        if skill.lower() in desc_lower:
            skills.append(skill)
    return list(set(skills))  # 去重
```

### 3. salary 解析与转换

招聘网站薪资格式多样，需要统一转换为 K/月。

```python
import re

def parse_salary(salary_text: str) -> tuple:
    """解析薪资文本，返回 (salary_min, salary_max)，单位K/月

    支持格式:
        "25-45K·14薪" → (25, 45)
        "30-60K" → (30, 60)
        "300-600元/天" → (约65, 130)  按22工作日转月薪
        "30万-60万/年" → (21, 43)     年薪转月薪（按14薪）
    """
    if not salary_text:
        return None, None

    # 匹配 "25-45K" 或 "25-45K·14薪"
    m = re.search(r'(\d+)-(\d+)\s*[Kk]', salary_text)
    if m:
        return int(m.group(1)), int(m.group(2))

    # 匹配 "300-600元/天"
    m = re.search(r'(\d+)-(\d+)\s*元/天', salary_text)
    if m:
        daily_min = int(m.group(1))
        daily_max = int(m.group(2))
        # 按22个工作日转月薪，再除1000转K
        return round(daily_min * 22 / 1000), round(daily_max * 22 / 1000)

    # 匹配 "30-60万/年"
    m = re.search(r'(\d+)-(\d+)\s*万', salary_text)
    if m:
        annual_min = int(m.group(1))  # 万
        annual_max = int(m.group(2))
        # 年薪转月薪（按14薪估算），万转K
        return round(annual_min * 10 / 14), round(annual_max * 10 / 14)

    return None, None
```

### 4. experience 提取

```python
def parse_experience(text: str) -> tuple:
    """解析经验要求，返回 (experience_min, experience_max)

    示例:
        "3-5年经验" → (3, 5)
        "5年以上" → (5, 99)
        "不限" → (0, 99)
    """
    if not text:
        return None, None

    m = re.search(r'(\d+)-(\d+)\s*年', text)
    if m:
        return int(m.group(1)), int(m.group(2))

    m = re.search(r'(\d+)\s*年以上', text)
    if m:
        return int(m.group(1)), 99

    if "不限" in text or "应届" in text:
        return 0, 99

    return None, None
```

```python
def crawl_job_detail(url: str, source: str = "web") -> dict:
    """抓取单个岗位详情页"""
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                      "AppleWebKit/537.36 (KHTML, like Gecko) "
                      "Chrome/125.0.0.0 Safari/537.36",
    }

    try:
        resp = requests.get(url, headers=headers, timeout=15)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, "html.parser")

        # ===== 提取字段 =====
        # 注意：以下选择器需根据实际目标网站调整
        job_title = soup.select_one(".job-title").get_text(strip=True)
        company_name = soup.select_one(".company-name").get_text(strip=True)
        salary_text = soup.select_one(".salary").get_text(strip=True)
        location = parse_location(soup.select_one(".location").get_text(strip=True))
        experience_text = soup.select_one(".experience").get_text(strip=True)
        education = parse_education(soup.select_one(".education").get_text(strip=True))
        job_description = soup.select_one(".job-description").get_text(strip=True)

        # 派生字段
        salary_min, salary_max = parse_salary(salary_text)
        experience_min, experience_max = parse_experience(experience_text)
        skills = extract_skills(job_description)

        return {
            "job_title": job_title,
            "company_name": company_name,
            "salary_min": salary_min,
            "salary_max": salary_max,
            "location": location,
            "experience_min": experience_min,
            "experience_max": experience_max,
            "education": education,
            "skills": skills,
            "job_description": job_description,
            "source": source,
            "source_url": url,
            "collected_at": datetime.now().isoformat(),
        }

    except Exception as e:
        print(f"抓取失败 [{url}]: {e}")
        return None


def crawl_job_list(base_url: str, keywords: list, max_count: int = 50) -> list:
    """批量抓取岗位列表"""
    all_jobs = []
    seen_titles = set()

    for keyword in keywords:
        if len(all_jobs) >= max_count:
            break

        # 搜索列表页
        search_url = f"{base_url}/search?q={quote_plus(keyword)}"
        try:
            resp = requests.get(search_url, timeout=15)
            soup = BeautifulSoup(resp.text, "html.parser")

            # 提取详情页链接（选择器需根据网站调整）
            links = soup.select(".job-card a")
            for link in links:
                if len(all_jobs) >= max_count:
                    break

                detail_url = link.get("href")
                if not detail_url:
                    continue

                job = crawl_job_detail(detail_url, source="web_crawler")
                if job:
                    # 去重：基于岗位标题前20字符
                    dedup_key = job["job_title"][:20]
                    if dedup_key in seen_titles:
                        continue
                    seen_titles.add(dedup_key)
                    all_jobs.append(job)

                time.sleep(0.5)  # 礼貌延迟

        except Exception as e:
            print(f"搜索失败 [{keyword}]: {e}")
            continue

    return all_jobs


def save_to_file(jobs: list, output_path: str):
    """保存采集结果到文件"""
    data = {
        "collected_at": datetime.now().isoformat(),
        "source": "web_crawler",
        "total_count": len(jobs),
        "jobs": jobs,
    }
    with open(output_path, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
    print(f"采集完成，共 {len(jobs)} 条，已保存到: {output_path}")
```

## 六、采集注意事项

### 1. 数据质量要求

- **job_description 必须完整**：包含"岗位职责"+"任职要求"两部分，越全越好
- **skills 提取不要硬编码太多**：如果抓不到 skills 字段，留空数组 `[]` 即可，系统会用LLM从description推断
- **薪资单位统一为 K/月**：年薪制、日薪制必须转换
- **所有文本UTF-8编码**：避免乱码

### 2. 反爬与礼貌策略

```python
# 必须设置合理的请求间隔
time.sleep(0.5)  # 每次请求间隔至少0.5秒

# 使用真实的User-Agent
headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) ..."
}

# 遇到反爬验证码时停止，不要暴力突破
if "验证码" in resp.text or resp.status_code == 403:
    print("遇到反爬，停止采集")
    break
```

### 3. 去重规则

同一公司同一岗位不要重复采集：

```python
# 去重key：岗位标题前20字符 + 公司名称
```
